using System;

namespace WebApi2.Models
{
    public class DataAccessClass
    {
        static SqlConnection connect;
        public static void Connection()
        {
            
            string dbConnect = ConfigurationManager.ConnectionStrings["dbconnect"].ConnectionString;
            connect = new SqlConnection(dbConnect);
            connect.Open();
        }
        public  user GetUser (int id){
            Connection();
           
            SqlDataReader reader = null;
            string query = "select * from users where id=" + id + "";
            SqlCommand command = new SqlCommand(query,connect);
            reader = command.ExecuteReader();
            user u = null;
            while (reader.Read())
            {
                u = new user();
                u.id = Convert.ToInt32(reader.GetValue(0));
                u.fname = reader.GetValue(1).ToString();
                u.lname = reader.GetValue(2).ToString();
                u.emailid = reader.GetValue(3).ToString();
                u.phone = reader.GetValue(4).ToString();
            }
            connect.Close();
            return u;
        }

        public user PostUser(string email, string password)
        {
            Connection();

                u.emailid = reader.GetValue(3).ToString();
                u.phone = reader.GetValue(4).ToString();
            }
            connect.Close();
            return u;

        }
